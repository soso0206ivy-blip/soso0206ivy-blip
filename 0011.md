
## 陣列

通常用來存放

### 宣告、推入、取出

```javascript
// 宣告
var names = ["Jerry", "Mary", "Amy"];
console.log(names);
// 推入
names.push("John");
console.log(names);
// 取出
var name = names.pop();
console.log(names);
console.log(name);
```

### 排序

```javascript
var names = ["Jerry", "Mary", "Amy"];
console.log(names.sort());
console.log(names.sort(function(x, y) {
    if(x > y){
        return 1;
    } else {
        return -1;
    }
}));

names.forEach(function(item, index){
    console.log(`第 ${index+1} 位是 ${item}`);
});

console.log(names.sort(function(x, y) {
    if(x < y){
        return 1;
    } else {
        return -1;
    }
}));

var arr = [168, NaN, 200, "Hello", 88,"", undefined, "*", "JavaScript", null, "@", "2"];
console.log(arr.sort());
console.log(arr.sort(function(x, y) {
  if(x > y){
    return 1; // 正數時，後面的數放在前面
  } else {
    return -1 // 負數時，前面的數放在前面
  }
}))


```
