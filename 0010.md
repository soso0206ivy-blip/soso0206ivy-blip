
## 函數

### 常用數學函數

#### 四捨五入 (整數)

```javascript
console.log(Math.round(3.7));
console.log(Math.round(3.4));
console.log(Math.round(3.5));
console.log(Math.round(-3.5));
console.log(Math.round(3.167*100)/100);
console.log(Math.round(3.165*100)/100);
console.log(Math.round(3.164*100)/100);
console.log(Math.round(-3.165*100)/100);

// toFixed
console.log(3.165.toFixed(2));
```

#### 小於或等於指定數值的最大整數

```javascript
console.log(Math.floor(3.5));
console.log(Math.floor(-3.5));
```

#### 大於或等於指定數值的最小整數

```javascript
console.log(Math.ceil(3.4));
console.log(Math.ceil(3.5));
console.log(Math.ceil(-3.4));
console.log(Math.ceil(-3.5));
```

#### 無條件捨去小數

```javascript
console.log(Math.trunc(3.168));
```

#### 次方


```javascript
var height = 168 / 100;
var weight = 65;
console.log((weight / Math.pow(height,2)).toFixed(2));

// BMI值計算公式 : BMI = 體重(公斤) / 身高^2(公尺^2)
function calc(weight, height){
    height = height / 100;
    bmi = (weight / Math.pow(height,2)).toFixed(2);
    if (bmi <=16) {
        alert("太輕了，需要醫師諮詢!!");
    }else if (bmi > 16 && bmi <= 18.5) {
        alert("體重過輕!!");
    }else if (bmi > 18.5 && bmi <= 25) {
        alert("體重正常!!");
    }else if (bmi > 25 && bmi <= 35) {
        alert("體重過重!!");
    }else {
        alert("太重了，需要醫師諮詢!!");
    }
}

```

#### 絕對值

console.log(Math.abs(-5));

### 字串函數

#### 長度

```javascript
var str = "Hello, JavaScript!";
console.log(str.length);
```

#### 找位置

```javascript
var str = "Hello, JavaScript!";
var findStr = "Java";
console.log(str.indexOf(findStr));
```

#### 轉大寫

```javascript
var str = "Hello, JavaScript!";
console.log(str.toUpperCase());
```

#### 轉小寫

```javascript
var str = "Hello, JavaScript!";
console.log(str.toLowerCase());
```

#### 取字串

```javascript
// str.slice(start, end)
var str = "Hello, JavaScript!";
console.log(str.slice(7, 11));
console.log(str.slice(-7, -1));

// str.substr(start, length)
console.log(str.substr(-7, 6));

```

#### 取代字串

```javascript
// str.replace(finder, newStr)
var str = "Hello, JavaScript!";
console.log(str.replace("JavaScript", "World"));
console.log(str);
console.log(str.replace(/a/, "@"));

```

##### 正規表示式

參考 : https://regex101.com/

```javascript
var str = "Hello, JavaScript!";
console.log(str.replace(/a/, "@"));
console.log(str.replace(/a/g, "@"));
console.log(str.replace(/A/g, "@"));
console.log(str.replace(/A/ig, "@"));

```

### 日期函數

#### 建立日期

```javascript
var x = 1;
console.log(typeof(x));
x = 1.2;
console.log(typeof(x));
x = "1.2";
console.log(typeof(x));
x = true;
console.log(typeof(x));

var date;
date = new Date();
console.log(typeof(date));
console.log(date);
date = new Date("2023-05-01");
console.log(date);
date = new Date("2023-05-01 01:00:00");
console.log(date);

```

#### 現在時間 Timestamp (UTC 時間)

自 1970 年 1 月 1 日 00:00:00 UTC 至今所經過的毫秒數。

```javascript
console.log(Date.now());

```

#### 取西元年份 / 月份 / 日數

月份為 0 ~ 11 (一月 ~ 十二月)

```javascript
var date;
date = new Date();
console.log(date.getFullYear());
console.log(date.getMonth() + 1);
console.log(date.getDate());

```

### 練習一下

#### 計算年齡

輸入名字、出生年份，計算並輸出使用者的年齡。

```javascript
var date = new Date();
var year = date.getFullYear();
var name = prompt("你的名字：");
var birthYear = prompt("出生年份：");
var age = year - birthYear;

console.log(`${name} 歡迎，你現在 ${age} 歲。`);

```

#### 計算圓面積和周長

```javascript
var radius = prompt("半徑：");
var perimeter = 2 * Math.PI * radius; // 2*pi*r
var area = Math.PI * Math.pow(radius,2); // pi*r^2

console.log(`半徑：${radius}`);
console.log(`周長：${perimeter}`);
console.log(`面積：${area}`);

```
